Discrete Loomis-Whitney inequality

Discrete Loomis-Whitney inequality. Let \(d \ge 2\) be an integer, and let \(A\) be a finite subset of \(\mathbb{Z}^d\). For each coordinate index \(i \in \{1,\dots,d\}\), let \(\pi_i : \mathbb{Z}^d \to \mathbb{Z}^{d-1}\) be the projection obtained by deleting the \(i\)-th coordinate. Then \[ |A|^{d-1} \le \prod_{i=1}^d |\pi_i(A)|. \] Equivalently, \[ |A| \le \prod_{i=1}^d |\pi_i(A)|^{1/(d-1)}. \] This is the discrete Loomis-Whitney inequality.

Informal statement:

For any integer \(d \ge 2\) and finite \(A \subset \mathbb{Z}^d\), \(|A| \le \prod_{i=1}^d |\pi_i(A)|^{1/(d-1)}\).

Informal proof:

We proceed by induction on \(d\). The base case \(d=2\) is handled separately. For the inductive step, assume the claim for \(d-1\). We slice \(A\) by the first coordinate to obtain \(A_t := \{x' \in \mathbb{Z}^{d-1} : (t,x') \in A\}\). The slices are finite and pairwise disjoint, so \(|A| = \sum_t |A_t|\). Applying the inductive hypothesis to each slice and combining the bounds via a multilinear Hölder estimate gives the desired inequality. Raising both sides to the \(d-1\)-st power yields \(|A|^{d-1} \le \prod_{i=1}^d |\pi_i(A)|\).

# Discrete Loomis–Whitney inequality

For any integer `d ≥ 2` and finite `A ⊂ ℤ^d`,
`|A|^{d-1} ≤ ∏_{i=1}^d |π_i(A)|`.

We prove this by induction on `d`. The base case `d = 2` is elementary, and
the induction step uses a multilinear Hölder estimate on slices.
-/

/-- The shadow (coordinate projection) of a finite set `A ⊆ ℤ^{n+1}` obtained by
    deleting coordinate `i`. -/
def shadow {n : ℕ} (i : Fin (n + 1)) (A : Finset (Fin (n + 1) → ℤ)) : Finset (Fin n → ℤ) :=
  A.image (· ∘ i.succAbove)

/-- The discrete Loomis–Whitney inequality in dimension `n + 1`:
    `|A|^n ≤ ∏ i, |π_i(A)|` for all finite `A ⊆ ℤ^{n+1}`. -/
def Ineq (n : ℕ) : Prop :=
  ∀ A : Finset (Fin (n + 1) → ℤ),
    (A.card : ℝ) ^ n ≤ ∏ i : Fin (n + 1), ((shadow i A).card : ℝ)

/-- Slice of `A ⊆ ℤ^{n+1}` at coordinate `0 = t`. -/
def sliceAt {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    Finset (Fin (n + 1) → ℤ) :=
  A.filter (fun v => v 0 = t)

/-- Embed a slice into the lower-dimensional space by deleting coordinate `0`. -/
def sliceEmbed {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    Finset (Fin n → ℤ) :=
  (sliceAt A t).image (· ∘ (0 : Fin (n + 1)).succAbove)

/-- The set of active slice indices. -/
def sliceIndices {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) : Finset ℤ :=
  A.image (· 0)

/-- Deleting coordinate `0` is injective on elements sharing the same `0`-th coordinate. -/
lemma eraseCoord_zero_injOn_slice {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    Set.InjOn (· ∘ (0 : Fin (n + 1)).succAbove)
      ((sliceAt A t : Finset (Fin (n + 1) → ℤ)) : Set (Fin (n + 1) → ℤ)) := by
  intro a ha b hb heq
  simp only [sliceAt, Finset.coe_filter, Set.mem_setOf_eq, Finset.mem_coe] at ha hb
  funext i
  by_cases hi : i = 0
  · subst hi; exact ha.2.trans hb.2.symm
  · have : i = (0 : Fin (n + 1)).succAbove (Fin.pred i hi) := by
      simp [Fin.succAbove_zero, Fin.succ_pred]
    rw [this]
    exact congr_fun heq (Fin.pred i hi)

/-- The cardinality of an embedded slice equals the cardinality of the original slice. -/
lemma sliceEmbed_card {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    (sliceEmbed A t).card = (sliceAt A t).card :=
  Finset.card_image_of_injOn (eraseCoord_zero_injOn_slice A t)

/-- Slicing partitions `A`: `|A| = ∑_{t ∈ T} |A_t|`. -/
lemma sum_slice_card {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) :
    ∑ t ∈ sliceIndices A, (sliceAt A t).card = A.card := by
  simp only [sliceAt, sliceIndices]
  rw [← Finset.card_eq_sum_card_fiberwise (fun x hx => Finset.mem_image_of_mem _ hx)]

/-- Each embedded slice is a subset of the shadow along coordinate `0`. -/
lemma sliceEmbed_subset_shadow {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    sliceEmbed A t ⊆ shadow 0 A := by
  unfold sliceEmbed shadow sliceAt
  exact Finset.image_subset_image (Finset.filter_subset _ _)

/-- The cardinality of each slice is bounded by the shadow along coordinate `0`. -/

    (sliceAt A t).card ≤ (shadow 0 A).card := by

**Base case**: The Loomis–Whitney inequality in dimension 2.
    For `A ⊆ ℤ²`, `|A| ≤ |π₁(A)| · |π₂(A)|`.
-/
theorem base_case : Ineq 1 := by
  intro A;
  norm_cast;
  -- The projection map from A to the product of the shadows is injective.
  have h_inj : Function.Injective (fun v : Fin 2 → ℤ => ((v 1 : ℤ), (v 0 : ℤ))) := by
    exact fun v w h => by ext i; fin_cases i <;> aesop;
  -- Since the projection map is injective, the cardinality of A is less than or equal to the cardinality of the image of A under the projection map.
  have h_card_le : A.card ≤ (Finset.image (fun v : Fin 2 → ℤ => (v 1, v 0)) A).card := by
    rw [ Finset.card_image_of_injective _ h_inj ];
  -- The image of A under the projection map is a subset of the product of the shadows.
  have h_image_subset : Finset.image (fun v : Fin 2 → ℤ => (v 1, v 0)) A ⊆ (shadow 0 A).image (fun v : Fin 1 → ℤ => (v 0)) ×ˢ (shadow 1 A).image (fun v : Fin 1 → ℤ => (v 0)) := by
    simp +decide [ Finset.subset_iff ];
    unfold shadow; aesop;
  have := Finset.card_mono h_image_subset; simp_all +decide [ Fin.prod_univ_succ ] ;
  exact h_card_le.trans ( this.trans ( Nat.mul_le_mul ( Finset.card_image_le ) ( Finset.card_image_le ) ) )

  induction n with
  | zero => omega

def Ineq (n : ℕ) : Prop :=
  ∀ A : Finset (Fin (n + 1) → ℤ),
    (A.card : ℝ) ^ n ≤ ∏ i : Fin (n + 1), ((shadow i A).card : ℝ)

/-- Slice of `A ⊆ ℤ^{n+1}` at coordinate `0 = t`. -/
def sliceAt {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    Finset (Fin (n + 1) → ℤ) :=
  A.filter (fun v => v 0 = t)

/-- Embed a slice into the lower-dimensional space by deleting coordinate `0`. -/
def sliceEmbed {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    Finset (Fin n → ℤ) :=
  (sliceAt A t).image (· ∘ (0 : Fin (n + 1)).succAbove)

/-- The set of active slice indices. -/
def sliceIndices {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) : Finset ℤ :=
  A.image (· 0)

/-- Deleting coordinate `0` is injective on elements sharing the same `0`-th coordinate. -/
lemma eraseCoord_zero_injOn_slice {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    Set.InjOn (· ∘ (0 : Fin (n + 1)).succAbove)
      ((sliceAt A t : Finset (Fin (n + 1) → ℤ)) : Set (Fin (n + 1) → ℤ)) := by
  intro a ha b hb heq
  simp only [sliceAt, Finset.coe_filter, Set.mem_setOf_eq, Finset.mem_coe] at ha hb
  funext i
  by_cases hi : i = 0
  · subst hi; exact ha.2.trans hb.2.symm
  · have : i = (0 : Fin (n + 1)).succAbove (Fin.pred i hi) := by
      simp [Fin.succAbove_zero, Fin.succ_pred]
    rw [this]
    exact congr_fun heq (Fin.pred i hi)

/-- The cardinality of an embedded slice equals the cardinality of the original slice. -/
lemma sliceEmbed_card {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    (sliceEmbed A t).card = (sliceAt A t).card :=
  Finset.card_image_of_injOn (eraseCoord_zero_injOn_slice A t)

/-- Slicing partitions `A`: `|A| = ∑_{t ∈ T} |A_t|`. -/
lemma sum_slice_card {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) :
    ∑ t ∈ sliceIndices A, (sliceAt A t).card = A.card := by
  simp only [sliceAt, sliceIndices]
  rw [← Finset.card_eq_sum_card_fiberwise (fun x hx => Finset.mem_image_of_mem _ hx)]

/-- Each embedded slice is a subset of the shadow along coordinate `0`. -/
lemma sliceEmbed_subset_shadow {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    sliceEmbed A t ⊆ shadow 0 A := by
  unfold sliceEmbed shadow sliceAt
  exact Finset.image_subset_image (Finset.filter_subset _ _)

/-- The cardinality of each slice is bounded by the shadow along coordinate `0`. -/
lemma slice_card_le_shadow_zero {n : ℕ} (A : Finset (Fin (n + 1) → ℤ)) (t : ℤ) :
    (sliceAt A t).card ≤ (shadow 0 A).card := by
  rw [← sliceEmbed_card]
  exact Finset.card_le_card (sliceEmbed_subset_shadow A t)

/-
The shadows of slices sum to at most the global shadow size.
-/
lemma sum_shadow_slice_le {n : ℕ} (A : Finset (Fin (n + 2) → ℤ)) (j : Fin (n + 1)) :
    ∑ t ∈ sliceIndices A, (shadow j (sliceEmbed A t)).card ≤
    (shadow (Fin.succAbove 0 j) A).card := by
  -- We can map each element in the shadow of a slice to an element in the global shadow by restoring the coordinate `0`.
  have h_map : ∀ t ∈ sliceIndices A, (shadow j (sliceEmbed A t)).image (Fin.cons t) ⊆ shadow (Fin.succAbove 0 j) A := by
    intro t ht; intro; simp +decide [ shadow, sliceEmbed ] ;
    rintro x hx rfl; use x; simp_all +decide [ sliceAt ] ;
    ext i; induction i using Fin.inductionOn <;> aesop;
  refine' le_trans _ ( Finset.card_le_card <| show ( Finset.biUnion ( sliceIndices A ) fun t => image ( Fin.cons t ) ( shadow j ( sliceEmbed A t ) ) ) ⊆ shadow ( Fin.succAbove 0 j ) A from _ );
  · rw [ Finset.card_biUnion ];
    · exact Finset.sum_le_sum fun t ht => by rw [ Finset.card_image_of_injective _ fun x y hxy => by simpa using hxy ] ;
    · intro t ht u hu htu; simp_all +decide [ Finset.disjoint_left ] ;
      aesop;
  · exact Finset.biUnion_subset.mpr h_map

/-
The induction step: Ineq d implies Ineq (d + 1).
-/
theorem step (d : ℕ) (hd : 1 ≤ d) (ih : Ineq d) : Ineq (d + 1) := by
  intro A
  set T := sliceIndices A
  have hA : (A.card : ℝ) = ∑ t ∈ T, (sliceAt A t).card := by
    rw_mod_cast [ sum_slice_card ];
  -- By the induction hypothesis, we have $|B_t|^{d+1} \leq |\pi_0(A)| \cdot \prod_j |\pi_j(B_t)|$.
  have h_ind : ∀ t ∈ T, (sliceAt A t).card ^ (d + 1) ≤ (shadow 0 A).card * ∏ j : Fin (d + 1), (shadow j (sliceEmbed A t)).card := by
    intro t ht
    have h_ind_step : (sliceEmbed A t).card ^ d ≤ ∏ j : Fin (d + 1), (shadow j (sliceEmbed A t)).card := by
      convert ih ( sliceEmbed A t ) using 1;
      norm_cast;
    convert Nat.mul_le_mul ( slice_card_le_shadow_zero A t ) h_ind_step using 1 ; ring!;
    rw [ sliceEmbed_card ];
  -- Applying the multilinear Hölder inequality to the sum of the products.
  have h_holder : (∑ t ∈ T, (∏ j : Fin (d + 1), (shadow j (sliceEmbed A t)).card : ℝ) ^ (1 / (d + 1 : ℝ))) ^ (d + 1) ≤ (∏ j : Fin (d + 1), (∑ t ∈ T, ((shadow j (sliceEmbed A t)).card : ℝ)) : ℝ) := by
    have h_holder : ∀ (f : Fin (d + 1) → ℤ → ℝ), (∀ j t, 0 ≤ f j t) → (∑ t ∈ T, (∏ j : Fin (d + 1), f j t) ^ (1 / (d + 1 : ℝ))) ^ (d + 1) ≤ (∏ j : Fin (d + 1), (∑ t ∈ T, f j t)) := by
      intro f hf_nonneg
      have h_holder_step : ∀ (t : ℤ), t ∈ T → (∏ j : Fin (d + 1), f j t) ^ (1 / (d + 1 : ℝ)) ≤ (∏ j : Fin (d + 1), (∑ t ∈ T, f j t)) ^ (1 / (d + 1 : ℝ)) * (∏ j : Fin (d + 1), (f j t / (∑ t ∈ T, f j t))) ^ (1 / (d + 1 : ℝ)) := by
        intro t ht; rw [ ← Real.mul_rpow ( Finset.prod_nonneg fun _ _ => Finset.sum_nonneg fun _ _ => hf_nonneg _ _ ) ( Finset.prod_nonneg fun _ _ => div_nonneg ( hf_nonneg _ _ ) ( Finset.sum_nonneg fun _ _ => hf_nonneg _ _ ) ) ] ;
        by_cases h : ∏ j : Fin ( d + 1 ), ∑ t ∈ T, f j t = 0 <;> simp_all +decide [ Finset.prod_eq_zero_iff, Finset.sum_eq_zero_iff_of_nonneg ];
        · obtain ⟨ j, hj ⟩ := h; simp_all +decide [ Finset.prod_eq_zero ( Finset.mem_univ j ) ] ;

      -- Applying the AM-GM inequality to the sum of the products.
      have h_am_gm : (∑ t ∈ T, (∏ j : Fin (d + 1), (f j t / (∑ t ∈ T, f j t))) ^ (1 / (d + 1 : ℝ))) ≤ 1 := by
        have h_am_gm : ∀ (t : ℤ), t ∈ T → (∏ j : Fin (d + 1), (f j t / (∑ t ∈ T, f j t))) ^ (1 / (d + 1 : ℝ)) ≤ (∑ j : Fin (d + 1), (f j t / (∑ t ∈ T, f j t))) / (d + 1) := by
          intro t ht; have := @Real.geom_mean_le_arith_mean;
          simpa [ Finset.sum_div _ _ _ ] using this Finset.univ ( fun _ => 1 ) ( fun j => f j t / ∑ t ∈ T, f j t ) ( fun _ _ => zero_le_one ) ( by simpa using by positivity ) ( fun _ _ => div_nonneg ( hf_nonneg _ _ ) ( Finset.sum_nonneg fun _ _ => hf_nonneg _ _ ) );
        refine le_trans ( Finset.sum_le_sum h_am_gm ) ?_;
        rw [ ← Finset.sum_div _ _ _, div_le_iff₀ ] <;> norm_cast <;> norm_num;
        rw [ Finset.sum_comm ];
        exact le_trans ( Finset.sum_le_sum fun _ _ => show ∑ x ∈ T, f _ x / ∑ t ∈ T, f _ t ≤ 1 from by rw [ ← Finset.sum_div _ _ _ ] ; exact div_self_le_one _ ) ( by norm_num );

    exact h_holder _ fun _ _ => Nat.cast_nonneg _;
  -- Applying the induction hypothesis to each term in the sum, we get:
  have h_sum_ind : (∑ t ∈ T, (sliceAt A t).card : ℝ) ≤ (shadow 0 A).card ^ (1 / (d + 1 : ℝ)) * (∑ t ∈ T, (∏ j : Fin (d + 1), (shadow j (sliceEmbed A t)).card : ℝ) ^ (1 / (d + 1 : ℝ))) := by
    have h_sum_ind : ∀ t ∈ T, (sliceAt A t).card ≤ (shadow 0 A).card ^ (1 / (d + 1 : ℝ)) * (∏ j : Fin (d + 1), (shadow j (sliceEmbed A t)).card : ℝ) ^ (1 / (d + 1 : ℝ)) := by
      intro t ht
      specialize h_ind t ht
      have h_root : ((sliceAt A t).card : ℝ) ^ (d + 1) ≤ (shadow 0 A).card * (∏ j : Fin (d + 1), (shadow j (sliceEmbed A t)).card : ℝ) := by
        exact_mod_cast h_ind;
      rw [ ← Real.mul_rpow ( by positivity ) ( by exact Finset.prod_nonneg fun _ _ => Nat.cast_nonneg _ ) ] ; exact le_trans ( by rw [ ← Real.rpow_natCast, ← Real.rpow_mul ( by positivity ), Nat.cast_add_one, mul_one_div_cancel ( by positivity ), Real.rpow_one ] ) ( Real.rpow_le_rpow ( by positivity ) h_root ( by positivity ) ) ;
    simpa only [ Finset.mul_sum _ _ _ ] using Finset.sum_le_sum h_sum_ind;
  -- Combining the inequalities from h_sum_ind and h_holder, we get:
  have h_combined : (∑ t ∈ T, (sliceAt A t).card : ℝ) ^ (d + 1) ≤ (shadow 0 A).card * (∏ j : Fin (d + 1), (∑ t ∈ T, ((shadow j (sliceEmbed A t)).card : ℝ)) : ℝ) := by
    refine le_trans ( pow_le_pow_left₀ ( Finset.sum_nonneg fun _ _ => Nat.cast_nonneg _ ) h_sum_ind _ ) ?_;
    rw [ mul_pow ];
    exact mul_le_mul ( by rw [ ← Real.rpow_natCast, ← Real.rpow_mul ( Nat.cast_nonneg _ ), Nat.cast_add_one, div_mul_cancel₀ _ ( by positivity ), Real.rpow_one ] ) h_holder ( by exact pow_nonneg ( Finset.sum_nonneg fun _ _ => Real.rpow_nonneg ( Finset.prod_nonneg fun _ _ => Nat.cast_nonneg _ ) _ ) _ ) ( by positivity );
  -- Applying the inequality from sum_shadow_slice_le to each term in the product.
  have h_prod_le : (∏ j : Fin (d + 1), (∑ t ∈ T, ((shadow j (sliceEmbed A t)).card : ℝ)) : ℝ) ≤ (∏ j : Fin (d + 1), ((shadow (Fin.succAbove 0 j) A).card : ℝ)) := by
    exact Finset.prod_le_prod ( fun _ _ => Finset.sum_nonneg fun _ _ => Nat.cast_nonneg _ ) fun _ _ => mod_cast sum_shadow_slice_le A _;
  convert h_combined.trans ( mul_le_mul_of_nonneg_left h_prod_le <| Nat.cast_nonneg _ ) using 1;
  · exact hA.symm ▸ by norm_cast;
  · rw [ Fin.prod_univ_succAbove ]
